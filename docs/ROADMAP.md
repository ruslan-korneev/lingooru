    quality: int,  # 0-5, –≥–¥–µ 5 = –∏–¥–µ–∞–ª—å–Ω–æ
    repetitions: int,
    easiness: float,
    interval: int
) -> tuple[int, float, int]:
    """
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç (–Ω–æ–≤—ã–µ_–ø–æ–≤—Ç–æ—Ä–µ–Ω–∏—è, –Ω–æ–≤—ã–π_easiness, –Ω–æ–≤—ã–π_–∏–Ω—Ç–µ—Ä–≤–∞–ª_–≤_–¥–Ω—è—Ö)

        return 0, easiness, 1

    # –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ easiness factor
    new_easiness = max(1.3, easiness + 0.1 - (5 - quality) * (0.08 + (5 - quality) * 0.02))

    # –†–∞—Å—á—ë—Ç –∏–Ω—Ç–µ—Ä–≤–∞–ª–∞

    elif repetitions == 1:

    else:

    return repetitions + 1, new_easiness, new_interval
